            name=tick\ticknum,
            inner xsep=0pt,
            },

        ticklabel style={
            font=\scriptsize,
            /pgf/number format/fixed,
            /pgf/number format/precision=2
            },
        xmin=\xmin,
        xmax=\xmax,
        enlargelimits=false,
    }
